// SIMULAÇÃO DE UM FILA COM MÚLTIPLOS COMANDOS //

INÍCIO

	DEFINIR VARIÁVEL ULTIMO = 10.

	DEFINIR VARIÁVEL FILA = UMA LISTA DOS NÚMEROS DE 0 À 10.

	FAÇA:

		EXIBA "EXISTEM {TAMANHO DE FILA} CLIENTES NA FILA"

		EXIBA "FILA ATUAL {FILA}"

		EXIBA "DIGITE F PARA ADICIONAR UM CLIENTE AO FIM DA FILA,"		
	
		EXIBA "OU A PARA REALIZAR O ATENDIMENTO. S PARA SAIR."
	
		DEFINIR VARIÁVEL OPERACAO.

		OPERACAO RECEBE O SEU VALOR DO USUÁRIO.

		DEFINIR VARIÁVEL x = 0.

		ENQUANTO x < TAMANHO DE OPERACAO, FAÇA:

			SE OPERACAO[x] == 'A' OU OPERACAO[x] == 'a':

				SE O TAMANHO DE FILA > 0:
				
					ATENDIDO = O PRIMEIRO ELEMENTO DE FILA E O REMOVE.

					EXIBA "CLIENTE {ATENDIDO} ATENDIDO"

				FIM SE:

				CASO CONTRÁRIO:

					EXIBA "FILA VAZIA! NINGUÉM PARA ATENDER"

				FIM CASO CONTRÁRIO:
			
			FIM SE:	
			
			SE OPERACAO[x] == 'F' OU OPERACAO[x] == 'f':

				ULTIMO += 1

				ADICIONAR ULTIMO NA ÚLTIMA POSIÇÃO DE FILA

			FIM SE:

			SE OPERACAO[x] == 'S' OU OPERACAO[x] == 's':

				EXIBA VARIÁVEL FILA.

				FINALIZAR OPERAÇÃO

			FIM SE:

			CASO CONTRÁRIO:

				EXIBA "OPERAÇÃO INVÁLIDA! DIGITE APENAS F, A OU S!"

			FIM CASO CONTRÁRIO:

			x += 1

		FIM ENQUANTO:


		

	FIM FAÇA:

FIM
