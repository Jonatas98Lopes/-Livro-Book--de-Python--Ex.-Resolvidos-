	DEFINIR OBJETO PILHA = [] <- (LISTA VAZIA)

	DEFINIR VARIÁVEL LEITOR (LERÁ UM CONJUNTO DE PARÊNTESES)

	EXIBA "DIGITE O CONJUNTO DE PARÊNTESES"

	LEITOR RECEBE SEU VALOR DO USUÁRIO

	DEFINIR VARIÁVEL x = 0.

	DEFINIR VARIÁVEL ADICIONA = 1.

	ENQUANTO x < TAMANHO DE LEITOR, FAÇA:
		
		SE LEITOR[x] == "(":
	
			PILHA RECEBE ADICIONA EM SUA ÚLTIMA POSIÇÃO. 

			ADICIONA += 1
 
		FIM SE:

		SE LEITOR[x] == ")" E TAMANHO DE PILHA > 0:

			REMOVER ÚLTIMO ELEMENTO DE PILHA.

			

		FIM SE:

		CASO CONTRÁRIO:

			EXIBA "ERRO"
			FINALIZAR OPERAÇÃO.

		FIM CASO CONTRÁRIO:		

		x += 1

	FIM ENQUANTO:

	CASO CONTRÁRIO (ENQUANTO):

		SE TAMANHO DE PILHA == 0:
			
			EXIBA "OK"

		FIM SE:

		CASO CONTRÁRIO:

			EXIBA "ERRO".

		FIM CASO CONTRÁRIO:

	FIM CASO CONTRÁRIO:
